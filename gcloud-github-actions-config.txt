gcloud iam workload-identity-pools create "github-actions-pool" \
  --project="gacs-playground" \
  --location="global" \
  --display-name="GitHub Actions Pool"

gcloud iam workload-identity-pools providers create-oidc "github-actions-provider" \
  --project="gacs-playground" \
  --location="global" \
  --workload-identity-pool="github-actions-pool" \
  --display-name="GitHub Actions Provider" \
  --attribute-mapping="google.subject=assertion.sub,attribute.repository=assertion.repository" \
  --attribute-condition="attribute.repository=='salgadoguiller/github-actions'"
  --issuer-uri="https://token.actions.githubusercontent.com"

gcloud iam service-accounts create github-actions \
  --project="gacs-playground" \
  --display-name="GitHub Actions"

gcloud projects add-iam-policy-binding gacs-playground \
  --member="serviceAccount:github-actions@gacs-playground.iam.gserviceaccount.com" \
  --role="roles/run.admin"

gcloud projects add-iam-policy-binding gacs-playground \
  --member="serviceAccount:github-actions@gacs-playground.iam.gserviceaccount.com" \
  --role="roles/iam.serviceAccountUser"

gcloud projects add-iam-policy-binding gacs-playground \
  --member="serviceAccount:github-actions@gacs-playground.iam.gserviceaccount.com" \
  --role="roles/artifactregistry.admin"
